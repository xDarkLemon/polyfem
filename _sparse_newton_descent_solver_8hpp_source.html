103 virtual bool compute_update_direction(
104 ProblemType &objFunc,
105 const TVector &x,
106 const TVector &grad,
107 TVector &direction) override
108 {
109 if (this->descent_strategy == 2)
110 {
111 direction = -grad;
112 return true;
113 }
114
115 {
116 POLYFEM_SCOPED_TIMER("assembly time", this->assembly_time);
117
118 if (this->descent_strategy == 1)
119 objFunc.set_project_to_psd(true);
120 else if (this->descent_strategy == 0)
121 objFunc.set_project_to_psd(false);
122 else
123 assert(false);
124
125 objFunc.hessian(x, hessian);
126
127 if (reg_weight > 0)
128 {
129 hessian += reg_weight * polyfem::utils::sparse_identity(hessian.rows(), hessian.cols());
130 }
131 }
132
133 {
134 POLYFEM_SCOPED_TIMER("linear solve", this->inverting_time);
135 // TODO: get the correct size
136 linear_solver->analyzePattern(hessian, hessian.rows());
137
138 try
139 {
140 linear_solver->factorize(hessian);
141 }
142 catch (const std::runtime_error &err)
143 {
144 increase_descent_strategy();
145 // warn if using gradient descent
146 polyfem::logger().log(
147 this->descent_strategy == 2 ? spdlog::level::warn : spdlog::level::debug,
148 "Unable to factorize Hessian: \"{}\"; reverting to {}",
149 err.what(), this->descent_strategy_name());
150 // polyfem::write_sparse_matrix_csv("problematic_hessian.csv", hessian);
151 return compute_update_direction(objFunc, x, grad, direction);
152 }
153
154 linear_solver->solve(-grad, direction); // H Δx = -g
155 }
156
157 // gradient descent, check descent direction
158 const double residual = (hessian * direction + grad).norm(); // H Δx + g = 0
159 if (std::isnan(residual))
160 {
161 increase_descent_strategy();
162 polyfem::logger().log(
163 this->descent_strategy == 2 ? spdlog::level::warn : spdlog::level::debug,
164 "nan linear solve residual {} (||∇f||={}); reverting to {}",
165 residual, grad.norm(), this->descent_strategy_name());
166 return compute_update_direction(objFunc, x, grad, direction);
167 }
168 else if (residual > std::max(1e-8 * grad.norm(), 1e-5))
169 {
170 increase_descent_strategy();
171 polyfem::logger().log(
172 this->descent_strategy == 2 ? spdlog::level::warn : spdlog::level::debug,
173 "large linear solve residual {} (||∇f||={}); reverting to {}",
174 residual, grad.norm(), this->descent_strategy_name());
175 return compute_update_direction(objFunc, x, grad, direction);
176 }
177 else
178 {
179 polyfem::logger().trace("linear solve residual {}", residual);
180 }
181
182 // do this check here because we need to repeat the solve without resetting reg_weight
183 if (grad.dot(direction) >= 0)
184 {
185 increase_descent_strategy();
186 polyfem::logger().log(
187 this->descent_strategy == 2 ? spdlog::level::warn : spdlog::level::debug,
188 "[{}] direction is not a descent direction (Δx⋅g={}≥0); reverting to {}",
189 name(), direction.dot(grad), descent_strategy_name());
190 return compute_update_direction(objFunc, x, grad, direction);
191 }
192
193 json info;
194 linear_solver->getInfo(info);
195 internal_solver_info.push_back(info);
196
197 reg_weight /= reg_weight_dec;
198 if (reg_weight < reg_weight_min)
199 reg_weight = 0;
200
201 return true;
202 }
