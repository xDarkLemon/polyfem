78bool polyfem::utils::read_matrix_ascii(const std::string &path, Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> &mat)
79{
80 std::fstream file;
81 file.open(path.c_str());
82
83 if (!file.good())
84 {
85 logger().error("Failed to open file: {}", path);
86 file.close();
87
88 return false;
89 }
90
91 std::string s;
92 std::vector<std::vector<T>> matrix;
93
94 while (getline(file, s))
95 {
96 if (s.empty())
97 continue;
98 std::stringstream input(s);
99 T temp;
100 matrix.emplace_back();
101
102 std::vector<T> &currentLine = matrix.back();
103
104 while (input >> temp)
105 currentLine.push_back(temp);
106 }
107
108 if (!igl::list_to_matrix(matrix, mat))
109 {
110 logger().error("list to matrix error");
111 file.close();
112
113 return false;
114 }
115

543Eigen::VectorXd polyfem::utils::flatten(const Eigen::MatrixXd &X)
544{
545 Eigen::VectorXd x(X.size());
546 for (int i = 0; i < X.rows(); ++i)
547 {
548 for (int j = 0; j < X.cols(); ++j)
549 {
550 x(i * X.cols() + j) = x(i);
551 }
552 }
553 assert(X(0, 0) == x(0));
554 assert(X.cols() <= 1 || X(0, 1) == x(1));
555 return x;
