248 {
249 Eigen::MatrixXd displacement_grad(size(), size());
250
251 assert(displacement.cols() == 1);
252
253 all.resize(local_pts.rows(), all_size);
254
255 ElementAssemblyValues vals;
256 vals.compute(el_id, size() == 3, local_pts, bs, gbs);
257
258 for (long p = 0; p < local_pts.rows(); ++p)
259 {
260 compute_diplacement_grad(size(), bs, vals, local_pts, p, displacement, displacement_grad);
261
262 const Eigen::MatrixXd def_grad = Eigen::MatrixXd::Identity(size(), size()) + displacement_grad;
263 const Eigen::MatrixXd FmT = def_grad.inverse().transpose();
264 // const double J = def_grad.determinant();
265
266 double lambda, mu;

275
276 all.row(p) = fun(stress_tensor);

291 // Compute ∫ ½μ (tr(FᵀF) - 3 - 2ln(J)) + ½λ ln²(J) du
292 template <typename T>
293 T NeoHookeanElasticity::compute_energy_aux(const ElementAssemblyValues &vals, const Eigen::MatrixXd &displacement, const QuadratureVector &da) const
294 {
295 typedef Eigen::Matrix<T, Eigen::Dynamic, 1> AutoDiffVect;
296 typedef Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> AutoDiffGradMat;
297
298 assert(displacement.cols() == 1);
299
300 const int n_pts = da.size();
301
302 Eigen::Matrix<double, Eigen::Dynamic, 1> local_dispv(vals.basis_values.size() * size(), 1);
303 local_dispv.setZero();
304 for (size_t i = 0; i < vals.basis_values.size(); ++i)
305 {
306 const auto &bs = vals.basis_values[i];
307 for (size_t ii = 0; ii < bs.global.size(); ++ii)
308 {
309 for (int d = 0; d < size(); ++d)
310 {
311 local_dispv(i * size() + d) += bs.global[ii].val * displacement(bs.global[ii].index * size() + d);
312 }
313 }
314 }
315
316 DiffScalarBase::setVariableCount(local_dispv.rows());
317 AutoDiffVect local_disp(local_dispv.rows(), 1);
318 T energy = T(0.0);
319
320 const AutoDiffAllocator<T> allocate_auto_diff_scalar;
321
322 for (long i = 0; i < local_dispv.rows(); ++i)
323 {
324 local_disp(i) = allocate_auto_diff_scalar(i, local_dispv(i));
325 }
326
327 AutoDiffGradMat def_grad(size(), size());
328
329 for (long p = 0; p < n_pts; ++p)
330 {
331 for (long k = 0; k < def_grad.size(); ++k)
332 def_grad(k) = T(0);
333
334 for (size_t i = 0; i < vals.basis_values.size(); ++i)
335 {
336 const auto &bs = vals.basis_values[i];
337 const Eigen::Matrix<double, Eigen::Dynamic, 1, 0, 3, 1> grad = bs.grad.row(p);
338 assert(grad.size() == size());
339
340 for (int d = 0; d < size(); ++d)
341 {
342 for (int c = 0; c < size(); ++c)
343 {
344 def_grad(d, c) += grad(c) * local_disp(i * size() + d);
345 }
346 }
347 }
348
349 AutoDiffGradMat jac_it(size(), size());
350 for (long k = 0; k < jac_it.size(); ++k)
351 jac_it(k) = T(vals.jac_it[p](k));
352 def_grad = def_grad * jac_it;
353
354 //Id + grad d
355 for (int d = 0; d < size(); ++d)
356 def_grad(d, d) += T(1);
357
358 double lambda, mu;
359 params_.lambda_mu(vals.quadrature.points.row(p), vals.val.row(p), vals.element_id, lambda, mu);
360
361 const T log_det_j = log(polyfem::utils::determinant(def_grad));
362 const T val = mu / 2 * ((def_grad.transpose() * def_grad).trace() - size() - 2 * log_det_j) + lambda / 2 * log_det_j * log_det_j;
363
364 energy += val * da(p);
365 }
366 return energy;
367 }

371 {

400 template <int n_basis, int dim>
401 void NeoHookeanElasticity::compute_energy_aux_gradient_fast(const ElementAssemblyValues &vals, const Eigen::MatrixXd &displacement, const QuadratureVector &da, Eigen::Matrix<double, Eigen::Dynamic, 1> &G_flattened) const
402 {
403
404 assert(displacement.cols() == 1);
405
406 const int n_pts = da.size();
407
408 Eigen::Matrix<double, n_basis, dim> local_disp(vals.basis_values.size(), size());
409 local_disp.setZero();
410 for (size_t i = 0; i < vals.basis_values.size(); ++i)
411 {
412 const auto &bs = vals.basis_values[i];
413 for (size_t ii = 0; ii < bs.global.size(); ++ii)
414 {
415 for (int d = 0; d < size(); ++d)
416 {
417 local_disp(i, d) += bs.global[ii].val * displacement(bs.global[ii].index * size() + d);
418 }
419 }
420 }
421
422 Eigen::Matrix<double, dim, dim> def_grad(size(), size());
423
424 Eigen::Matrix<double, n_basis, dim> G(vals.basis_values.size(), size());
425 G.setZero();
426
427 for (long p = 0; p < n_pts; ++p)
428 {
429 Eigen::Matrix<double, n_basis, dim> grad(vals.basis_values.size(), size());
430
431 for (size_t i = 0; i < vals.basis_values.size(); ++i)
432 {
433 grad.row(i) = vals.basis_values[i].grad.row(p);
434 }
435
436 Eigen::Matrix<double, dim, dim> jac_it = vals.jac_it[p];
437
438 //Id + grad d
439 def_grad = local_disp.transpose() * grad * jac_it + Eigen::Matrix<double, dim, dim>::Identity(size(), size());
440
441 const double J = def_grad.determinant();
442 const double log_det_j = log(J);
443
444 Eigen::Matrix<double, dim, dim> delJ_delF(size(), size());
445 delJ_delF.setZero();
446
447 if (dim == 2)
448 {
449
450 delJ_delF(0, 0) = def_grad(1, 1);
451 delJ_delF(0, 1) = -def_grad(1, 0);
452 delJ_delF(1, 0) = -def_grad(0, 1);
453 delJ_delF(1, 1) = def_grad(0, 0);
454 }
455
456 else if (dim == 3)
457 {
458
459 Eigen::Matrix<double, dim, 1> u(def_grad.rows());
460 Eigen::Matrix<double, dim, 1> v(def_grad.rows());
461 Eigen::Matrix<double, dim, 1> w(def_grad.rows());
462
463 u = def_grad.col(0);
464 v = def_grad.col(1);
465 w = def_grad.col(2);
466
467 delJ_delF.col(0) = cross<dim>(v, w);
468 delJ_delF.col(1) = cross<dim>(w, u);
469 delJ_delF.col(2) = cross<dim>(u, v);
470 }
471
472 double lambda, mu;
473 params_.lambda_mu(vals.quadrature.points.row(p), vals.val.row(p), vals.element_id, lambda, mu);
474
475 Eigen::Matrix<double, n_basis, dim> delF_delU = grad * jac_it;
476
477 Eigen::Matrix<double, dim, dim> gradient_temp = mu * def_grad - mu * (1 / J) * delJ_delF + lambda * log_det_j * (1 / J) * delJ_delF;
478 Eigen::Matrix<double, n_basis, dim> gradient = delF_delU * gradient_temp.transpose();
479
480 double val = mu / 2 * ((def_grad.transpose() * def_grad).trace() - size() - 2 * log_det_j) + lambda / 2 * log_det_j * log_det_j;
481
482 G.noalias() += gradient * da(p);
483 }
484
485 Eigen::Matrix<double, dim, n_basis> G_T = G.transpose();
486
487 constexpr int N = (n_basis == Eigen::Dynamic) ? Eigen::Dynamic : n_basis * dim;
488 Eigen::Matrix<double, N, 1> temp(Eigen::Map<Eigen::Matrix<double, N, 1>>(G_T.data(), G_T.size()));
489 G_flattened = temp;
490 }

494 {
495 assert(displacement.cols() == 1);
496

498 const int n_pts = da.size();
499
500 Eigen::Matrix<double, n_basis, dim> local_disp(vals.basis_values.size(), size());
501 local_disp.setZero();

505 for (size_t ii = 0; ii < bs.global.size(); ++ii)

526
527 //Id + grad d
528 def_grad = local_disp.transpose() * grad * jac_it + Eigen::Matrix<double, dim, dim>::Identity(size(), size());
529
530 const double J = def_grad.determinant();
531 double log_det_j = log(J);
532
533 Eigen::Matrix<double, dim, dim> delJ_delF(size(), size());
534 delJ_delF.setZero();
535 Eigen::Matrix<double, dim * dim, dim * dim> del2J_delF2(size() * size(), size() * size());
536 del2J_delF2.setZero();
537
538 if (dim == 2)
539 {
540 delJ_delF(0, 0) = def_grad(1, 1);
541 delJ_delF(0, 1) = -def_grad(1, 0);
542 delJ_delF(1, 0) = -def_grad(0, 1);
543 delJ_delF(1, 1) = def_grad(0, 0);
544
545 del2J_delF2(0, 3) = 1;
546 del2J_delF2(1, 2) = -1;
547 del2J_delF2(2, 1) = -1;
548 del2J_delF2(3, 0) = 1;
549 }
550 else if (size() == 3)
551 {
552 Eigen::Matrix<double, dim, 1> u(def_grad.rows());
553 Eigen::Matrix<double, dim, 1> v(def_grad.rows());
554 Eigen::Matrix<double, dim, 1> w(def_grad.rows());
555
556 u = def_grad.col(0);
557 v = def_grad.col(1);
558 w = def_grad.col(2);
559
560 delJ_delF.col(0) = cross<dim>(v, w);
561 delJ_delF.col(1) = cross<dim>(w, u);
562 delJ_delF.col(2) = cross<dim>(u, v);
563
564 del2J_delF2.template block<dim, dim>(0, 6) = hat<dim>(v);
565 del2J_delF2.template block<dim, dim>(6, 0) = -hat<dim>(v);
566 del2J_delF2.template block<dim, dim>(0, 3) = -hat<dim>(w);
567 del2J_delF2.template block<dim, dim>(3, 0) = hat<dim>(w);
568 del2J_delF2.template block<dim, dim>(3, 6) = -hat<dim>(u);
569 del2J_delF2.template block<dim, dim>(6, 3) = hat<dim>(u);
570 }
571
572 double lambda, mu;
573 params_.lambda_mu(vals.quadrature.points.row(p), vals.val.row(p), vals.element_id, lambda, mu);
574
575 Eigen::Matrix<double, dim * dim, dim *dim> id = Eigen::Matrix<double, dim * dim, dim * dim>::Identity(size() * size(), size() * size());
576
577 Eigen::Matrix<double, dim * dim, 1> g_j = Eigen::Map<const Eigen::Matrix<double, dim * dim, 1>>(delJ_delF.data(), delJ_delF.size());
578
579 Eigen::Matrix<double, dim * dim, dim *dim> hessian_temp = (mu * id) + (((mu + lambda * (1 - log_det_j)) / (J * J)) * (g_j * g_j.transpose())) + (((lambda * log_det_j - mu) / (J)) * del2J_delF2);
580
581 Eigen::Matrix<double, dim * dim, N> delF_delU_tensor(jac_it.size(), grad.size());
582
583 for (size_t i = 0; i < local_disp.rows(); ++i)
584 {
585 for (size_t j = 0; j < local_disp.cols(); ++j)
586 {
587 Eigen::Matrix<double, dim, dim> temp(size(), size());
588 temp.setZero();
589 temp.row(j) = grad.row(i);
590 temp = temp * jac_it;
591 Eigen::Matrix<double, dim * dim, 1> temp_flattened(Eigen::Map<Eigen::Matrix<double, dim * dim, 1>>(temp.data(), temp.size()));
592 delF_delU_tensor.col(i * size() + j) = temp_flattened;
593 }
594 }
595
596 Eigen::Matrix<double, N, N> hessian = delF_delU_tensor.transpose() * hessian_temp * delF_delU_tensor;
597
598 double val = mu / 2 * ((def_grad.transpose() * def_grad).trace() - size() - 2 * log_det_j) + lambda / 2 * log_det_j * log_det_j;
599
600 H += hessian * da(p);
601 }
602 }

604 void NeoHookeanElasticity::compute_dstress_dgradu_multiply_mat(const int el_id, const Eigen::MatrixXd &local_pts, const Eigen::MatrixXd &global_pts, const Eigen::MatrixXd &grad_u_i, const Eigen::MatrixXd &mat, Eigen::MatrixXd &stress, Eigen::MatrixXd &result) const
605 {
606 double lambda, mu;
607 params_.lambda_mu(local_pts, global_pts, el_id, lambda, mu);
608
609 Eigen::MatrixXd def_grad = Eigen::MatrixXd::Identity(grad_u_i.rows(), grad_u_i.cols()) + grad_u_i;
610 Eigen::MatrixXd FmT = def_grad.inverse().transpose();
611
612 stress = mu * (def_grad - FmT) + lambda * std::log(def_grad.determinant()) * FmT;
613 result = mu * mat + FmT * mat.transpose() * FmT * (mu - lambda * std::log(def_grad.determinant())) + lambda * (FmT.array() * mat.array()).sum() * FmT;
614 }
