216 void compute_diplacement_grad(const int size, const ElementBases &bs, const ElementAssemblyValues &vals, const Eigen::MatrixXd &local_pts, const int p, const Eigen::MatrixXd &displacement, Eigen::MatrixXd &displacement_grad)
217 {
218 assert(displacement.cols() == 1);
219
220 displacement_grad.setZero();
221
222 for (std::size_t j = 0; j < bs.bases.size(); ++j)
223 {
224 const Basis &b = bs.bases[j];
225 const auto &loc_val = vals.basis_values[j];
226
227 assert(bs.bases.size() == vals.basis_values.size());
228 assert(loc_val.grad.rows() == local_pts.rows());
229 assert(loc_val.grad.cols() == size);
230
231 for (int d = 0; d < size; ++d)
232 {
233 for (std::size_t ii = 0; ii < b.global().size(); ++ii)
234 {
235 displacement_grad.row(d) += b.global()[ii].val * loc_val.grad.row(p) * displacement(b.global()[ii].index * size + d);
236 }
237 }
238 }
239
240 displacement_grad = (displacement_grad * vals.jac_it[p]).eval();

243 double von_mises_stress_for_stress_tensor(const Eigen::MatrixXd &stress)
244 {
245 double von_mises_stress;
246
247 if (stress.rows() == 3)
248 {
249 von_mises_stress = 0.5 * (stress(0, 0) - stress(1, 1)) * (stress(0, 0) - stress(1, 1)) + 3.0 * stress(0, 1) * stress(1, 0);
250 von_mises_stress += 0.5 * (stress(2, 2) - stress(1, 1)) * (stress(2, 2) - stress(1, 1)) + 3.0 * stress(2, 1) * stress(2, 1);
251 von_mises_stress += 0.5 * (stress(2, 2) - stress(0, 0)) * (stress(2, 2) - stress(0, 0)) + 3.0 * stress(2, 0) * stress(2, 0);
252 }
253 else
254 {
255 // von_mises_stress = ( stress(0, 0) - stress(1, 1) ) * ( stress(0, 0) - stress(1, 1) ) + 3.0 * stress(0, 1) * stress(1, 0);
256 von_mises_stress = stress(0, 0) * stress(0, 0) - stress(0, 0) * stress(1, 1) + stress(1, 1) * stress(1, 1) + 3.0 * stress(0, 1) * stress(1, 0);
257 }
258
259 von_mises_stress = sqrt(fabs(von_mises_stress));
260
261 return von_mises_stress;

416 void ElasticityTensor::set_from_young_poisson(const double young, const double nu)
417 {
418 if (size_ == 2)
419 {
420 stifness_tensor_ << 1.0, nu, 0.0,
421 nu, 1.0, 0.0,
422 0.0, 0.0, (1.0 - nu) / 2.0;
423 stifness_tensor_ *= young / (1.0 - nu * nu);
424 }
425 else
426 {
427 assert(size_ == 3);
428 const double v = nu;
429 stifness_tensor_ << 1. - v, v, v, 0, 0, 0,
430 v, 1. - v, v, 0, 0, 0,
431 v, v, 1. - v, 0, 0, 0,
432 0, 0, 0, (1. - 2. * v) / 2., 0, 0,
433 0, 0, 0, 0, (1. - 2. * v) / 2., 0,
434 0, 0, 0, 0, 0, (1. - 2. * v) / 2.;
435 stifness_tensor_ *= young / ((1. + v) * (1. - 2. * v));
