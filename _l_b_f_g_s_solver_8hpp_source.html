94
95 virtual bool compute_update_direction(
96 ProblemType &objFunc,
97 const TVector &x,
98 const TVector &grad,
99 TVector &direction) override
100 {
101 if (this->descent_strategy == 2)
102 {
103 // Use gradient descent in the first iteration or if the previous iteration failed
104 direction = -grad;
105 }
106 else
107 {
108 // Update s and y
109 // s_{i+1} = x_{i+1} - x_i
110 // y_{i+1} = g_{i+1} - g_i
111 m_bfgs.add_correction(x - m_prev_x, grad - m_prev_grad);
112
113 // Recursive formula to compute d = -H * g
114 m_bfgs.apply_Hv(grad, -Scalar(1), direction);
115 }
116
117 m_prev_x = x;
118 m_prev_grad = grad;
119
120 return true;
121 }
122 };
