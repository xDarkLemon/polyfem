82 in_node_offset.head(num_vertex_nodes).setZero();

94 range.array() / node_per_prim + prim_offset;

97 range.unaryExpr([&](const int x) { return x % node_per_prim; });

99 prim_offset += num_prims;

186 void State::build_node_mapping()
187 {
188 if (!mesh->is_conforming())
189 return;
190 const int num_vertex_nodes = mesh_nodes->num_vertex_nodes();
191 const int num_edge_nodes = mesh_nodes->num_edge_nodes();
192 const int num_face_nodes = mesh_nodes->num_face_nodes();
193 const int num_cell_nodes = mesh_nodes->num_cell_nodes();

195 const int num_nodes = num_vertex_nodes + num_edge_nodes + num_face_nodes + num_cell_nodes;
196
197 const long n_vertices = num_vertex_nodes;
198 const int num_in_primitives = n_vertices + mesh->n_edges() + mesh->n_faces() + mesh->n_cells();
199 const int num_primitives = mesh->n_vertices() + mesh->n_edges() + mesh->n_faces() + mesh->n_cells();
200
201 Eigen::VectorXi in_node_to_in_primitive;
202 Eigen::VectorXi in_node_offset;

207 mesh->in_ordered_vertices(),
208 mesh->in_ordered_edges(),
209 mesh->in_ordered_faces(),
210 in_primitive_to_primitive);
211
212 const auto primitive_offset = [&](int node) {
213 if (mesh_nodes->is_vertex_node(node))
214 return 0;
215 else if (mesh_nodes->is_edge_node(node))
216 return mesh->n_vertices();
217 else if (mesh_nodes->is_face_node(node))
218 return mesh->n_vertices() + mesh->n_edges();
219 else if (mesh_nodes->is_cell_node(node))
220 return mesh->n_vertices() + mesh->n_edges() + mesh->n_faces();
221 throw std::runtime_error("Invalid node ID!");
222 };
223
224 std::vector<std::vector<int>> primitive_to_nodes(num_primitives);
225 const std::vector<int> &grouped_nodes = mesh_nodes->primitive_to_node();
226 int node_count = 0;
227 for (int i = 0; i < grouped_nodes.size(); i++)
228 {
229 int node = grouped_nodes[i];
230 assert(node < num_nodes);
231 if (node >= 0)
232 {
233 int primitive = mesh_nodes->node_to_primitive_gid().at(node) + primitive_offset(i);
234 assert(primitive < num_primitives);
235 primitive_to_nodes[primitive].push_back(node);
236 node_count++;
237 }
238 }
239 assert(node_count == num_nodes);
240
241 in_node_to_node.resize(num_nodes);
242 for (int i = 0; i < num_nodes; i++)
243 {
244 // input node id -> input primitive -> primitive -> node
245 const std::vector<int> &possible_nodes =
246 primitive_to_nodes[in_primitive_to_primitive[in_node_to_in_primitive[i]]];
247 assert(possible_nodes.size() > 0);
248 if (possible_nodes.size() > 1)
249 {
250#ifndef NDEBUG
251 // assert(mesh_nodes->is_edge_node(i)); // TODO: Handle P4+
252 for (int possible_node : possible_nodes)
253 assert(mesh_nodes->is_edge_node(possible_node)); // TODO: Handle P4+
254#endif
255
256 int e_id = in_primitive_to_primitive[in_node_to_in_primitive[i]] - mesh->n_vertices();
257 assert(e_id < mesh->n_edges());
258 RowVectorNd v0 = mesh_nodes->node_position(in_node_to_node[mesh->edge_vertex(e_id, 0)]);
259 RowVectorNd a = mesh_nodes->node_position(possible_nodes[0]);
260 RowVectorNd b = mesh_nodes->node_position(possible_nodes[1]);
261 // Assume possible nodes are ordered, so only need to check order of two nodes
262
263 // Input edges are sorted, so if a is closer to v0 then the order is correct
264 // otherwise the nodes are flipped.
265 assert(mesh->edge_vertex(e_id, 0) < mesh->edge_vertex(e_id, 1));
266 int offset = (a - v0).squaredNorm() < (b - v0).squaredNorm()
267 ? in_node_offset[i]
268 : (possible_nodes.size() - in_node_offset[i] - 1);
269 in_node_to_node[i] = possible_nodes[offset];
270 }
271 else
272 in_node_to_node[i] = possible_nodes[0];
273 }

333 void State::compute_mesh_size(const Mesh &mesh_in, const std::vector<ElementBases> &bases_in, const int n_samples)
334 {
335 Eigen::MatrixXd samples_simplex, samples_cube, mapped, p0, p1, p;
336
337 mesh_size = 0;
338 average_edge_length = 0;
339 min_edge_length = std::numeric_limits<double>::max();
340
341 if (!args["output"]["advanced"]["curved_mesh_size"])
342 {
343 mesh_in.get_edges(p0, p1);
344 p = p0 - p1;
345 min_edge_length = p.rowwise().norm().minCoeff();
346 average_edge_length = p.rowwise().norm().mean();
347 mesh_size = p.rowwise().norm().maxCoeff();
348
349 logger().info("hmin: {}", min_edge_length);
350 logger().info("hmax: {}", mesh_size);
351 logger().info("havg: {}", average_edge_length);
352
353 return;
354 }
355
356 if (mesh_in.is_volume())
357 {
358 EdgeSampler::sample_3d_simplex(n_samples, samples_simplex);
359 EdgeSampler::sample_3d_cube(n_samples, samples_cube);
360 }
361 else
362 {
363 EdgeSampler::sample_2d_simplex(n_samples, samples_simplex);
364 EdgeSampler::sample_2d_cube(n_samples, samples_cube);
365 }
366
367 int n = 0;
368 for (size_t i = 0; i < bases_in.size(); ++i)
369 {
370 if (mesh_in.is_polytope(i))
371 continue;
372 int n_edges;
373
374 if (mesh_in.is_simplex(i))
375 {
376 n_edges = mesh_in.is_volume() ? 6 : 3;
377 bases_in[i].eval_geom_mapping(samples_simplex, mapped);
378 }
379 else
380 {
381 n_edges = mesh_in.is_volume() ? 12 : 4;
382 bases_in[i].eval_geom_mapping(samples_cube, mapped);
383 }
384
385 for (int j = 0; j < n_edges; ++j)
386 {
387 double current_edge = 0;
388 for (int k = 0; k < n_samples - 1; ++k)
389 {
390 p0 = mapped.row(j * n_samples + k);
391 p1 = mapped.row(j * n_samples + k + 1);
392 p = p0 - p1;
393
394 current_edge += p.norm();
395 }
396
397 mesh_size = std::max(current_edge, mesh_size);
398 min_edge_length = std::min(current_edge, min_edge_length);
399 average_edge_length += current_edge;
400 ++n;
401 }
402 }
403
404 average_edge_length /= n;
405
406 logger().info("hmin: {}", min_edge_length);
407 logger().info("hmax: {}", mesh_size);
408 logger().info("havg: {}", average_edge_length);
409 }

478 void compute_integral_constraints(
479 const Mesh3D &mesh,
480 const int n_bases,
481 const std::vector<ElementBases> &bases,
482 const std::vector<ElementBases> &gbases,
483 Eigen::MatrixXd &basis_integrals)
484 {
485 if (!mesh.is_volume())
486 {
487 logger().error("Works only on volumetric meshes!");
488 return;
489 }
490 assert(mesh.is_volume());
491
492 basis_integrals.resize(n_bases, 9);
493 basis_integrals.setZero();
494 Eigen::MatrixXd rhs(n_bases, 9);
495 rhs.setZero();
496
497 const int n_elements = mesh.n_elements();
498 for (int e = 0; e < n_elements; ++e)
499 {
500 // if (mesh.is_polytope(e)) {
501 // continue;
502 // }
503 // ElementAssemblyValues vals = values[e];
504 // const ElementAssemblyValues &gvals = gvalues[e];
505 ElementAssemblyValues vals;
506 vals.compute(e, mesh.is_volume(), bases[e], gbases[e]);
507
508 // Computes the discretized integral of the PDE over the element
509 const int n_local_bases = int(vals.basis_values.size());
510 for (int j = 0; j < n_local_bases; ++j)
511 {
512 const AssemblyValues &v = vals.basis_values[j];
513 const double integral_100 = (v.grad_t_m.col(0).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
514 const double integral_010 = (v.grad_t_m.col(1).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
515 const double integral_001 = (v.grad_t_m.col(2).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
516
517 const double integral_110 = ((vals.val.col(1).array() * v.grad_t_m.col(0).array() + vals.val.col(0).array() * v.grad_t_m.col(1).array()) * vals.det.array() * vals.quadrature.weights.array()).sum();
518 const double integral_011 = ((vals.val.col(2).array() * v.grad_t_m.col(1).array() + vals.val.col(1).array() * v.grad_t_m.col(2).array()) * vals.det.array() * vals.quadrature.weights.array()).sum();
519 const double integral_101 = ((vals.val.col(0).array() * v.grad_t_m.col(2).array() + vals.val.col(2).array() * v.grad_t_m.col(0).array()) * vals.det.array() * vals.quadrature.weights.array()).sum();
520
521 const double integral_200 = 2 * (vals.val.col(0).array() * v.grad_t_m.col(0).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
522 const double integral_020 = 2 * (vals.val.col(1).array() * v.grad_t_m.col(1).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
523 const double integral_002 = 2 * (vals.val.col(2).array() * v.grad_t_m.col(2).array() * vals.det.array() * vals.quadrature.weights.array()).sum();
524
525 const double area = (v.val.array() * vals.det.array() * vals.quadrature.weights.array()).sum();
526
527 for (size_t ii = 0; ii < v.global.size(); ++ii)
528 {
529 basis_integrals(v.global[ii].index, 0) += integral_100 * v.global[ii].val;
530 basis_integrals(v.global[ii].index, 1) += integral_010 * v.global[ii].val;
531 basis_integrals(v.global[ii].index, 2) += integral_001 * v.global[ii].val;
532
533 basis_integrals(v.global[ii].index, 3) += integral_110 * v.global[ii].val;
534 basis_integrals(v.global[ii].index, 4) += integral_011 * v.global[ii].val;
535 basis_integrals(v.global[ii].index, 5) += integral_101 * v.global[ii].val;
536
537 basis_integrals(v.global[ii].index, 6) += integral_200 * v.global[ii].val;
538 basis_integrals(v.global[ii].index, 7) += integral_020 * v.global[ii].val;
539 basis_integrals(v.global[ii].index, 8) += integral_002 * v.global[ii].val;
540
541 rhs(v.global[ii].index, 6) += -2.0 * area * v.global[ii].val;
542 rhs(v.global[ii].index, 7) += -2.0 * area * v.global[ii].val;
543 rhs(v.global[ii].index, 8) += -2.0 * area * v.global[ii].val;
544 }
545 }
546 }
547
548 basis_integrals -= rhs;
549 }
