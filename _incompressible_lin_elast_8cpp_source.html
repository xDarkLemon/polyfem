30 IncompressibleLinearElasticityDispacement::assemble(const ElementAssemblyValues &vals, const int i, const int j, const QuadratureVector &da) const
31 {
32 // 2mu (epsi : epsj)
33 assert(size_ == 2 || size_ == 3);
34
35 Eigen::Matrix<double, Eigen::Dynamic, 1, 0, 9, 1> res(size() * size());
36 res.setZero();
37
38 const Eigen::MatrixXd &gradi = vals.basis_values[i].grad_t_m;
39 const Eigen::MatrixXd &gradj = vals.basis_values[j].grad_t_m;
40
41 Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> epsi(size(), size());
42 Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> epsj(size(), size());
43
44 for (long p = 0; p < gradi.rows(); ++p)
45 {
46
47 double lambda, mu;
48 params_.lambda_mu(vals.quadrature.points.row(p), vals.val.row(p), vals.element_id, lambda, mu);
49
50 for (int di = 0; di < size(); ++di)
51 {
52 epsi.setZero();
53 epsi.row(di) = gradi.row(p);
54 epsi = ((epsi + epsi.transpose()) / 2).eval();
55 for (int dj = 0; dj < size(); ++dj)
56 {
57 epsj.setZero();
58 epsj.row(dj) = gradj.row(p);
59 epsj = ((epsj + epsj.transpose()) / 2.0).eval();
60
61 res(dj * size() + di) += 2 * mu * (epsi.array() * epsj.array()).sum() * da(p);
62 }
63 }
64 }
65
66 return res;
67 }
