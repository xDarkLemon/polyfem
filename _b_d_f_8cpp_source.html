94 {
95 const Eigen::VectorXd sum_x_prev = weighted_sum_x_prevs();

97 const double beta = betas(x_prevs.size() - 1);
98
99 x_prevs.push_front(x); // x_prev() = x
100 v_prevs.push_front((x_prev() - sum_x_prev) / (beta * dt()));

103 if (x_prevs.size() > num_steps)
