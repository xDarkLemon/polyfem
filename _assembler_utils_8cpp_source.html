82 hooke_linear_elasticity_.assemble(is_volume, n_basis, bases, gbases, cache, stiffness);

89 return;
90 else if (assembler == "NeoHookean")

572 void AssemblerUtils::merge_mixed_matrices(
573 const int n_bases, const int n_pressure_bases, const int problem_dim, const bool add_average,
574 const StiffnessMatrix &velocity_stiffness, const StiffnessMatrix &mixed_stiffness, const StiffnessMatrix &pressure_stiffness,
575 StiffnessMatrix &stiffness)
576 {
577 assert(velocity_stiffness.rows() == velocity_stiffness.cols());
578 assert(velocity_stiffness.rows() == n_bases * problem_dim);
579
580 assert(mixed_stiffness.size() == 0 || mixed_stiffness.rows() == n_bases * problem_dim);
581 assert(mixed_stiffness.size() == 0 || mixed_stiffness.cols() == n_pressure_bases);
582
583 assert(pressure_stiffness.size() == 0 || pressure_stiffness.rows() == n_pressure_bases);
584 assert(pressure_stiffness.size() == 0 || pressure_stiffness.cols() == n_pressure_bases);
585
586 const int avg_offset = add_average ? 1 : 0;
587
588 std::vector<Eigen::Triplet<double>> blocks;
589 blocks.reserve(velocity_stiffness.nonZeros() + 2 * mixed_stiffness.nonZeros() + pressure_stiffness.nonZeros() + 2 * avg_offset * velocity_stiffness.rows());
590
591 for (int k = 0; k < velocity_stiffness.outerSize(); ++k)
592 {
593 for (StiffnessMatrix::InnerIterator it(velocity_stiffness, k); it; ++it)
594 {
595 blocks.emplace_back(it.row(), it.col(), it.value());
596 }
597 }
598
599 for (int k = 0; k < mixed_stiffness.outerSize(); ++k)
600 {
601 for (StiffnessMatrix::InnerIterator it(mixed_stiffness, k); it; ++it)
602 {
603 blocks.emplace_back(it.row(), n_bases * problem_dim + it.col(), it.value());
604 blocks.emplace_back(it.col() + n_bases * problem_dim, it.row(), it.value());
605 }
606 }
607
608 for (int k = 0; k < pressure_stiffness.outerSize(); ++k)
609 {
610 for (StiffnessMatrix::InnerIterator it(pressure_stiffness, k); it; ++it)
611 {
612 blocks.emplace_back(n_bases * problem_dim + it.row(), n_bases * problem_dim + it.col(), it.value());
613 }
614 }
615
616 if (add_average)
617 {
618 const double val = 1.0 / n_pressure_bases;
619 for (int i = 0; i < n_pressure_bases; ++i)
620 {
621 blocks.emplace_back(n_bases * problem_dim + i, n_bases * problem_dim + n_pressure_bases, val);
622 blocks.emplace_back(n_bases * problem_dim + n_pressure_bases, n_bases * problem_dim + i, val);
623 }
624 }
625
626 stiffness.resize(n_bases * problem_dim + n_pressure_bases + avg_offset, n_bases * problem_dim + n_pressure_bases + avg_offset);
627 stiffness.setFromTriplets(blocks.begin(), blocks.end());
628 stiffness.makeCompressed();

631 // Eigen::saveMarket(stiffness, "stiffness.txt");
632 // Eigen::saveMarket(velocity_stiffness, "velocity_stiffness.txt");
633 // Eigen::saveMarket(mixed_stiffness, "mixed_stiffness.txt");
634 // Eigen::saveMarket(pressure_stiffness, "pressure_stiffness.txt");
635 }
