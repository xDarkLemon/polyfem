40 // Constructs a list of unique faces represented in a given mesh (V,T)
41 //
42 // Inputs:
43 // T: #T × 4 matrix of indices of tet corners
44 // Outputs:
45 // F: #F × 3 list of faces in no particular order
46 template <typename DerivedT, typename DerivedF>
47 void get_faces(
48 const Eigen::MatrixBase<DerivedT> &T,
49 Eigen::PlainObjectBase<DerivedF> &F)
50 {
51 assert(T.rows() == 4);
52 assert(T.cols() >= 1);
53
54 Eigen::MatrixXi BF, OF;
55 igl::boundary_facets(T, BF);
56 igl::oriented_facets(T, OF); // boundary facets + duplicated interior faces
57 assert((OF.rows() + BF.rows()) % 2 == 0);
58 const int num_faces = (OF.rows() + BF.rows()) / 2;
59 F.resize(num_faces, 3);
60 F.topRows(BF.rows()) = BF;
61 std::unordered_set<std::vector<int>, HashVector> processed_faces;
62 for (int fi = 0; fi < BF.rows(); fi++)
63 {
64 processed_faces.insert(sort_face(BF.row(fi)));
65 }
66
67 for (int fi = 0; fi < OF.rows(); fi++)
68 {
69 std::vector<int> sorted_face = sort_face(OF.row(fi));
70 const auto iter = processed_faces.find(sorted_face);
71 if (iter == processed_faces.end())
72 {
73 F.row(processed_faces.size()) = OF.row(fi);
74 processed_faces.insert(sorted_face);
75 }
76 }
77
78 assert(F.rows() == processed_faces.size());
79 }

163 std::unique_ptr<Mesh> Mesh::create(const std::string &path, const bool non_conforming)
164 {
165 if (!std::filesystem::exists(path))
166 {
167 logger().error(path.empty() ? "No mesh provided!" : "Mesh file does not exist: {}", path);
168 return nullptr;
169 }
170
171 std::string lowername = path;
172
173 std::transform(lowername.begin(), lowername.end(), lowername.begin(), ::tolower);
174 if (StringUtils::endswith(lowername, ".hybrid"))
175 {
176 std::unique_ptr<Mesh> mesh;
177 if (non_conforming)
178 mesh = std::make_unique<NCMesh3D>();
179 else
180 mesh = std::make_unique<CMesh3D>();
181 if (mesh->load(path))
182 {
183 //TODO add in_ordered_vertices_, in_ordered_edges_, in_ordered_faces_
184 return mesh;
185 }
186 }
187 else if (StringUtils::endswith(lowername, ".msh"))
188 {
189 Eigen::MatrixXd vertices;
190 Eigen::MatrixXi cells;
191 std::vector<std::vector<int>> elements;
192 std::vector<std::vector<double>> weights;
193 std::vector<int> body_ids;
194
195 if (!MshReader::load(path, vertices, cells, elements, weights, body_ids))
196 return nullptr;
197
198 std::unique_ptr<Mesh> mesh;
199 if (vertices.cols() == 2)
200 if (non_conforming)
201 mesh = std::make_unique<NCMesh2D>();
202 else
203 mesh = std::make_unique<CMesh2D>();
204 else if (non_conforming)
205 mesh = std::make_unique<NCMesh3D>();
206 else
207 mesh = std::make_unique<CMesh3D>();
208
209 mesh->build_from_matrices(vertices, cells);
210 // Only tris and tets
211 if ((vertices.cols() == 2 && cells.cols() == 3) || (vertices.cols() == 3 && cells.cols() == 4))
212 {
213 mesh->attach_higher_order_nodes(vertices, elements);
214 mesh->cell_weights_ = weights;
215
216 // TODO, not clear?
217 }
218
219 for (const auto &w : weights)
220 {
221 if (!w.empty())
222 {
223 mesh->is_rational_ = true;
224 break;
225 }
226 }
227
228 mesh->set_body_ids(body_ids);
229
230 return mesh;
231 }
232 else
233 {
234 GEO::Mesh tmp;
235 if (GEO::mesh_load(path, tmp))
236 {
237 return create(tmp, non_conforming);
238 }
239 }
240 logger().error("Failed to load mesh: {}", path);
241 return nullptr;

244 std::unique_ptr<Mesh> Mesh::create(
245 const Eigen::MatrixXd &vertices, const Eigen::MatrixXi &cells, const bool non_conforming)
246 {
247 const int dim = vertices.cols();
248
249 std::unique_ptr<Mesh> mesh;
250 if (dim == 2)
251 {
252 if (non_conforming)
253 mesh = std::make_unique<NCMesh2D>();
254 else
255 mesh = std::make_unique<CMesh2D>();
256 }
257 else
258 {
259 assert(dim == 3);
260 if (non_conforming)
261 mesh = std::make_unique<NCMesh3D>();
262 else
263 mesh = std::make_unique<CMesh3D>();
264 }
265
266 mesh->build_from_matrices(vertices, cells);
267
268 mesh->in_ordered_vertices_ = Eigen::VectorXi::LinSpaced(vertices.rows(), 0, vertices.rows() - 1);
269 assert(mesh->in_ordered_vertices_[0] == 0);
270 assert(mesh->in_ordered_vertices_[1] == 1);
271 assert(mesh->in_ordered_vertices_[2] == 2);
272 assert(mesh->in_ordered_vertices_[mesh->in_ordered_vertices_.size() - 1] == vertices.rows() - 1);
273
274 if (dim == 2)
275 {
276 std::unordered_set<std::pair<int, int>, HashPair> edges;
277 for (int f = 0; f < cells.rows(); ++f)
278 {
279 for (int lv = 0; lv < cells.cols(); ++lv)
280 {
281 const int v0 = cells(f, lv);
282 const int v1 = cells(f, (lv + 1) % cells.cols());
283 edges.emplace(std::pair<int, int>(std::min(v0, v1), std::max(v0, v1)));
284 }
285 }
286 mesh->in_ordered_edges_.resize(edges.size(), 2);
287 int index = 0;
288 for (auto it = edges.begin(); it != edges.end(); ++it)
289 {
290 mesh->in_ordered_edges_(index, 0) = it->first;
291 mesh->in_ordered_edges_(index, 1) = it->second;
292 ++index;
293 }
294
295 assert(mesh->in_ordered_edges_.size() > 0);
296
297 mesh->in_ordered_faces_.resize(0, 0);
298 }
299 else
300 {
301 if (cells.rows() == 4)
302 {
303 get_faces(cells, mesh->in_ordered_faces_);
304 igl::edges(mesh->in_ordered_faces_, mesh->in_ordered_edges_);
305 }
306 // else TODO
307 }
308
309 return mesh;
310 }

384 {
385 boundary_ids_.resize(is_volume() ? n_faces() : n_edges());
386
387 std::ifstream file(path);
388
389 std::string line;

391 while (std::getline(file, line))
392 {
393 std::istringstream iss(line);
394 int v;
395 iss >> v;
396 boundary_ids_[bindex] = v;
397

400
