202 template <typename T>
203 T OgdenElasticity::compute_energy_aux(const ElementAssemblyValues &vals, const Eigen::MatrixXd &displacement, const QuadratureVector &da) const
204 {
205 typedef Eigen::Matrix<T, Eigen::Dynamic, 1> AutoDiffVect;
206 typedef Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, 0, 3, 3> AutoDiffGradMat;
207
208 assert(displacement.cols() == 1);
209
210 const int n_pts = da.size();
211
212 Eigen::Matrix<double, Eigen::Dynamic, 1> local_dispv(vals.basis_values.size() * size(), 1);
213 local_dispv.setZero();
214 for (size_t i = 0; i < vals.basis_values.size(); ++i)
215 {
216 const auto &bs = vals.basis_values[i];
217 for (size_t ii = 0; ii < bs.global.size(); ++ii)
218 {
219 for (int d = 0; d < size(); ++d)
220 {
221 local_dispv(i * size() + d) += bs.global[ii].val * displacement(bs.global[ii].index * size() + d);
222 }
223 }
224 }
225
226 DiffScalarBase::setVariableCount(local_dispv.rows());
227 AutoDiffVect local_disp(local_dispv.rows(), 1);
228 T energy = T(0.0);
229
230 const AutoDiffAllocator<T> allocate_auto_diff_scalar;
231
232 for (long i = 0; i < local_dispv.rows(); ++i)
233 {
234 local_disp(i) = allocate_auto_diff_scalar(i, local_dispv(i));
235 }
236
237 AutoDiffGradMat def_grad(size(), size());
238
239 for (long p = 0; p < n_pts; ++p)
240 {
241 for (long k = 0; k < def_grad.size(); ++k)
242 def_grad(k) = T(0);
243
244 for (size_t i = 0; i < vals.basis_values.size(); ++i)
245 {
246 const auto &bs = vals.basis_values[i];
247 const Eigen::Matrix<double, Eigen::Dynamic, 1, 0, 3, 1> grad = bs.grad.row(p);
248 assert(grad.size() == size());
249
250 for (int d = 0; d < size(); ++d)
251 {
252 for (int c = 0; c < size(); ++c)
253 {
254 def_grad(d, c) += grad(c) * local_disp(i * size() + d);
255 }
256 }
257 }
258
259 AutoDiffGradMat jac_it(size(), size());
260 for (long k = 0; k < jac_it.size(); ++k)
261 jac_it(k) = T(vals.jac_it[p](k));
262
263 def_grad = def_grad * jac_it;
264
265 //Id + grad d
266 for (int d = 0; d < size(); ++d)
267 def_grad(d, d) += T(1);
268
269 Eigen::Matrix<T, Eigen::Dynamic, 1, 0, 3, 1> eigs;
270
271 if (size() == 2)
272 autogen::eigs_2d<T>(def_grad, eigs);
273 else if (size() == 3)
274 autogen::eigs_3d<T>(def_grad, eigs);
275 else
276 assert(false);
277
278 const T J = polyfem::determinant(def_grad);
279 const T Jdenom = pow(J, -1. / size());
280
281 for (long i = 0; i < eigs.size(); ++i)
282 eigs(i) = eigs(i) * Jdenom;
283
284 auto val = T(0);
285 for (long N = 0; N < alphas_.size(); ++N)
286 {
287 auto tmp = T(-size());
288 const double alpha = alphas_(N);
289 const double mu = mus_(N);
290
291 for (long i = 0; i < eigs.size(); ++i)
292 tmp += pow(eigs(i), alpha);
293
294 val += 2 * mu / (alpha * alpha) * tmp;
295 }
296
297 // std::cout<<val<<std::endl;
298
299 for (long N = 0; N < Ds_.size(); ++N)
300 {
301 const double D = Ds_(N);
302
303 val += 1. / D * pow(J - T(1), 2 * (N + 1));
304 }
305
306 energy += val * da(p);
307 }
308 // std::cout<<"\n\n------------\n\n\n"<<std::endl;
309 return energy;
310 }
