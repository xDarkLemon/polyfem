592 void transform_mesh_from_json(const json &transform, Eigen::MatrixXd &vertices)
593 {
594 const int dim = vertices.cols();
595 assert(dim == 2 || dim == 3);
596
597 // -----
598 // Scale
599 // -----
600
601 RowVectorNd scale;
602 if (transform["dimensions"].is_array()) // default is nullptr
603 {
604 VectorNd initial_dimensions =
605 (vertices.colwise().maxCoeff() - vertices.colwise().minCoeff()).cwiseAbs();
606 initial_dimensions =
607 (initial_dimensions.array() == 0).select(1, initial_dimensions);
608
609 scale = transform["dimensions"];
610 const int scale_size = scale.size();
611 scale.conservativeResize(dim);
612 if (scale_size < dim)
613 scale.tail(dim - scale_size).setZero();
614
615 scale.array() /= initial_dimensions.array();
616 }
617 else if (transform["scale"].is_number())
618 {
619 scale.setConstant(dim, transform["scale"].get<double>());
620 }
621 else
622 {
623 assert(transform["scale"].is_array());
624 scale = transform["scale"];
625 const int scale_size = scale.size();
626 scale.conservativeResize(dim);
627 if (scale_size < dim)
628 scale.tail(dim - scale_size).setZero();
629 }
630
631 vertices *= scale.asDiagonal();
632
633 // ------
634 // Rotate
635 // ------
636
637 // Rotate around the models origin NOT the bodies center of mass.
638 // We could expose this choice as a "rotate_around" field.
639 MatrixNd R = MatrixNd::Identity(dim, dim);
640 if (!transform["rotation"].is_null())
641 {
642 if (dim == 2)
643 {
644 if (transform["rotation"].is_number())
645 R = Eigen::Rotation2Dd(deg2rad(transform["rotation"].get<double>()))
646 .toRotationMatrix();
647 else
648 log_and_throw_error("Invalid 2D rotation; 2D rotations can only be a angle in degrees.");
649 }
650 else if (dim == 3)
651 {
652 R = to_rotation_matrix(transform["rotation"], transform["rotation_mode"]);
653 }
654 }
655
656 vertices *= R.transpose(); // (R*Vᵀ)ᵀ = V*Rᵀ
657
658 // ---------
659 // Translate
660 // ---------
661
662 RowVectorNd translation = transform["translation"];
663 const int translation_size = translation.size();
664 translation.conservativeResize(dim);
665 if (translation_size < dim)
666 translation.tail(dim - translation_size).setZero();
667
668 vertices.rowwise() += translation;
669 }

673 void append_selections(
674 const std::string &root_path,
675 const json &new_selections,
676 const Selection::BBox &bbox,
677 const size_t &start_element_id,
678 const size_t &end_element_id,
679 std::vector<std::shared_ptr<Selection>> &selections)
680 {
681 if (new_selections.is_number_integer())
682 {
683 selections.push_back(std::make_shared<UniformSelection>(
684 new_selections.get<int>(), start_element_id, end_element_id));
685 }
686 else if (new_selections.is_string())
687 {
688 selections.push_back(std::make_shared<FileSelection>(
689 resolve_path(new_selections, root_path), start_element_id, end_element_id));
690 }
691 else if (new_selections.is_object())
692 {
693 selections.push_back(Selection::build(
694 new_selections, bbox, start_element_id, end_element_id));
695 }
696 else if (new_selections.is_array())
697 {
698 std::vector<json> new_selections_array = new_selections;
699 for (const json &s : new_selections_array)
700 {
701 selections.push_back(Selection::build(
702 s, bbox, start_element_id, end_element_id));
703 }
704 }
705 else if (!new_selections.is_null())
706 {
707 log_and_throw_error(fmt::format("Invalid selections: {}", new_selections));
708 }
